Enrollment:

CREATE TABLE STUDENT(
    S_ID int NOT NULL,
    S_NAME char(50) NOT NULL,
    S_MAJOR char(50) NOT NULL,
    PRIMARY KEY 	(S_ID)
    );

CREATE TABLE COURSE(
    C_DEPTCODE char(4) NOT NULL ,
    C_COURSENUM int NOT NULL,
    C_TITLE char(50) NOT NULL,
    C_CREDITHOURS int NOT NULL,
    PRIMARY KEY 	(C_DEPTCODE, C_COURSENUM )
    );

CREATE TABLE ENROLLMENT(
	ENROLLMENT_ID int NOT NULL AUTO_INCREMENT,
    STUDENT_ID int NOT NULL,
    COURSE_DEPTCODE char(4) NOT NULL ,
    COURSE_COURSENUM int NOT NULL,
    PRIMARY KEY 	(ENROLLMENT_ID),
    FOREIGN KEY (STUDENT_ID) REFERENCES STUDENT(S_ID) ON DELETE NO ACTION,
    FOREIGN KEY (COURSE_DEPTCODE) REFERENCES COURSE(C_DEPTCODE, C_COURSENUM ) ON DELETE NO ACTION,
    UNIQUE(ENROLLMENT_ID)
    );

    
drop  table ENROLLMENT, COURSE, STUDENT;
