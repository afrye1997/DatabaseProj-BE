CREATE TABLE STUDENT(
    S_ID int NOT NULL AUTO_INCREMENT,
    S_NAME char(50) NOT NULL,
    S_MAJOR char(50) NOT NULL,
    PRIMARY KEY 	(S_ID)
    );



CREATE TABLE COURSE(
    C_DEPTCODE int NOT NULL ,
    C_COURSENUM int NOT NULL,
    C_TITLE char(50) NOT NULL,
    C_CREDITHOURS int NOT NULL,
    PRIMARY KEY 	(C_DEPTCODE)
    );



CREATE TABLE ENROLLMENT(
    STUDENT_ID int NOT NULL ,
    COURSE_DEPTCODE int NOT NULL ,
    COURSECOURSENUM int NOT NULL,
    PRIMARY KEY 	(STUDENT_ID),
    FOREIGN KEY (STUDENT_ID) REFERENCES STUDENTS(S_ID) ON DELETE NO ACTION,
    FOREIGN KEY (COURSE_DEPTCODE) REFERENCES ENROLLMENT(C_DEPTCODE) ON DELETE NO ACTION
    );



CREATE TABLE ENROLLMENT(
	ENROLLMENT_ID int UNIQUE,
    STUDENT_ID int NOT NULL,
    COURSE_DEPTCODE int NOT NULL ,
    COURSE_COURSENUM int NOT NULL,
    PRIMARY KEY 	(ENROLLMENT_ID),
    FOREIGN KEY (STUDENT_ID) REFERENCES STUDENT(S_ID) ON DELETE NO ACTION,
    FOREIGN KEY (COURSE_DEPTCODE) REFERENCES COURSE(C_DEPTCODE) ON DELETE NO ACTION
    );
